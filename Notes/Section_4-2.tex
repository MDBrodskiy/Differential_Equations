      \begin{split}
      a_2(x)y''+a_1(x)y'+a(x)y=0\\

      u[y_1''+Py_1'+Qy_1]+y_1u''+(2y_1'+Py_1)u'=0\\
      y_1w'+(2y_1'+Py_1)w=0
    \end{split}
