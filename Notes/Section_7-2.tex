\begin{itemize}

  \item Some known values for inverse Laplace transforms, $\mathcal{L}^{-1}\{f(t)\}$, are shown:

  \begin{equation}
      \begin{split}
        1&=\mathcal{L}^{-1}\left\{ \frac{1}{s} \right\}\\
        t^n=\mathcal{L}^{-1}\left\{ \frac{n!}{s^{n+1}} \right\}\,\,\,\,&\,\,\,\, e^{at}=\mathcal{L}^{-1}\left\{ \frac{1}{s-a} \right\}\\
        \sin kt=\mathcal{L}^{-1}\left\{ \frac{k}{s^2+k^2} \right\}\,\,\,\,&\,\,\,\, \cos kt=\mathcal{L}^{-1}\left\{ \frac{s}{s^2+k^2} \right\}\\
        \sinh kt=\mathcal{L}^{-1}\left\{ \frac{k}{s^2-k^2} \right\}\,\,\,\,&\,\,\,\, \cosh kt=\mathcal{L}^{-1}\left\{ \frac{s}{s^2-k^2} \right\}\\
      \end{split}
      \label{1}
    \end{equation}

  \item If $f,f',\dots,f^{n-1}$ are continuous on $[0,\infty)$, and are of exponential order and if $f^{(n)}(t)$ is piecewise continuous on $[0,\infty)$, then \eqref{2} where $F(s)=\mathcal{L}\{f(t)\}

    \begin{equation}
      \mathcal{L}\{f^{(n)}(t)\}=s^nF(s)-s^{n-1}f(0)-s^{n-2}f'(0)-\dots-f^{(n-1)}(0)
      \label{2}
    \end{equation}

\end{itemize}
